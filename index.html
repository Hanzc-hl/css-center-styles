1. absolute + 负margin（子元素定宽）

          .parent {
            position: relative;
          }
        
          .child {
            position: absolute;
            top: 50%;
            left: 50%;
            margin-top: -40px;
            margin-left: -50px;
            width: 100px;
            height: 80px;
          }

2. absolute + calc（子元素定宽）

          .parent {
            position: relative;
          }
        
          .child {
            position: absolute;
            top: calc(50% - 40px);
            left: calc(50% - 50px);
            width: 100px;
            height: 80px;
          }

3. absolute + transform

          .parent {
            position: relative;
          }
        
          .child {
            position: absolute;
            top: 50%;
            left: 50%;
            transform: translate(-50%, -50%);
          }

4. absolute + margin:auto

          .parent {
            position: relative;
          }
        
          .child {
            position: absolute;
            top: 0;
            right: 0;
            bottom: 0;
            left: 0;
            margin: auto;
          }

5. flex + margin:auto

          .parent {
            display: flex;
          }
        
          .child {
            margin: auto;
          }

            display: flex;

7. grid

          .parent {
            display: grid;
            justify-content: center;
            align-items: center;
          }
        
          .child { 
            width: 100px;
            height: 80px;
          }

8. line-height

          .parent {
            line-height: 160px;

          }
